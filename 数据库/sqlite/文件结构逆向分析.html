# sqlite3 文件结构逆向分析

# 初始化数据结构，并插入数据

sqlite3 a.db

sqlite> .databases
main: /Users/ydl/a.db
sqlite> .table
test   test1
sqlite> .tables test
test
sqlite> .schema test
CREATE TABLE test (`id` int);
sqlite> .schema test1
CREATE TABLE test1 (`test` varchar(30));
sqlite> select *from test;
1
1234567
sqlite> select * from test1;
hello world
